  <script type="module">
    import { client as supabase } from '../../modules/core/supabase.js';
    import { initSettlement } from './settlement.js';

    (async () => {
      const { data: sessionRes, error } = await supabase.auth.getSession();
      if (error) {
        console.warn("세션 조회 에러:", error);
      }
      if (!sessionRes?.session) {
        console.warn("로그인 세션 없음 → /map으로 이동");
        location.replace("https://baikuk.com/map");
        return;
      }

      const user = sessionRes.session.user;
      if (!user?.id) {
        location.replace("https://baikuk.com/map");
        return;
      }

      const { data: staff, error: staffError } = await supabase
        .from("staff_profiles")
        .select("authority")
        .eq("user_id", user.id)
        .maybeSingle();

      if (staffError) {
        console.warn("권한 조회 실패:", staffError);
        return;
      }

      if (staff?.authority?.trim() === "직원") {
        console.warn("직원 권한 → /admin/performance 로 이동");
        location.replace("/admin/performance/");
      } else {
        await initSettlement();  // ✅ 바로 실행
      }
    })();
  </script>
